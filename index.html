import customtkinter as ctk import cv2 import mediapipe as mp import numpy as np from PIL import Image, ImageTk import pandas as pd from datetime import datetime import pyttsx3 import threading # --- 1. إعدادات المحركات الخلفية --- engine = pyttsx3.init() def speak(text): threading.Thread(target=lambda: (engine.say(text), engine.runAndWait())).start() mp_pose = mp.solutions.pose pose = mp_pose.Pose(min_detection_confidence=0.7, min_tracking_confidence=0.7) mp_drawing = mp.solutions.drawing_utils # --- 2. الكلاس الرئيسي للتطبيق --- class SmartPhysioFullApp(ctk.CTk): def __init__(self): super().__init__() # إعدادات النافذة self.title("نظام AI Physio Pro المتكامل") self.geometry("1300x800") ctk.set_appearance_mode("dark") # متغيرات التشغيل والقاعدة self.counter = 0 self.stage = None self.current_exercise = "الكوع" self.history_data = [] # --- بناء الواجهة الرسومية --- self.grid_columnconfigure(1, weight=1) self.grid_rowconfigure(0, weight=1) # الجانب الأيسر: التحكم والقائمة self.sidebar = ctk.CTkFrame(self, width=280, corner_radius=15) self.sidebar.grid(row=0, column=0, padx=20, pady=20, sticky="nsew") ctk.CTkLabel(self.sidebar, text="العيادة الذكية", font=("Arial", 26, "bold")).pack(pady=30) # أزرار التمارين self.btn_elbow = ctk.CTkButton(self.sidebar, text="تمرين ثني الكوع", height=45, command=lambda: self.select_ex("الكوع")) self.btn_elbow.pack(pady=10, padx=20) self.btn_shoulder = ctk.CTkButton(self.sidebar, text="تمرين رفع الكتف", height=45, command=lambda: self.select_ex("الكتف")) self.btn_shoulder.pack(pady=10, padx=20) # زر حفظ البيانات self.save_btn = ctk.CTkButton(self.sidebar, text="تصدير التقرير (Excel)", fg_color="#27ae60", hover_color="#2ecc71", command=self.export_to_excel) self.save_btn.pack(side="bottom", pady=40, padx=20) # الجانب الأيمن: الكاميرا والبيانات self.main_view = ctk.CTkFrame(self, corner_radius=15, fg_color="#121212") self.main_view.grid(row=0, column=1, padx=20, pady=20, sticky="nsew") self.video_display = ctk.CTkLabel(self.main_view, text="") self.video_display.pack(expand=True, fill="both", padx=15, pady=15) # شريط المعلومات السفلي self.bottom_bar = ctk.CTkFrame(self, height=150, corner_radius=15) self.bottom_bar.grid(row=1, column=0, columnspan=2, padx=20, pady=(0, 20), sticky="ew") self.angle_text = ctk.CTkLabel(self.bottom_bar, text="الزاوية: 0°", font=("Arial", 25)) self.angle_text.pack(side="left", expand=True) self.counter_text = ctk.CTkLabel(self.bottom_bar, text="0", font=("Arial", 70, "bold"), text_color="#3498db") self.counter_text.pack(side="left", expand=True) self.status_text = ctk.CTkLabel(self.bottom_bar, text="جاهز للبدء", font=("Arial", 22), text_color="#f1c40f") self.status_text.pack(side="left", expand=True) # بدء الكاميرا self.cap = cv2.VideoCapture(0) self.run_ai_loop() # --- المنطق البرمجي --- def select_ex(self, ex_name): self.current_exercise = ex_name self.counter = 0 self.stage = None speak(f"تم اختيار تمرين {ex_name}") def calc_angle(self, a, b, c): a, b, c = np.array(a), np.array(b), np.array(c) radians = np.arctan2(c[1]-b[1], c[0]-b[0]) - np.arctan2(a[1]-b[1], a[0]-b[0]) angle = np.abs(radians*180.0/np.pi) return 360-angle if angle > 180 else angle def export_to_excel(self): if not self.history_data: speak("لا توجد بيانات لحفظها") return df = pd.DataFrame(self.history_data, columns=['الوقت', 'نوع التمرين', 'العدة رقم']) filename = f"Physio_Report_{datetime.now().strftime('%Y%m%d_%H%M')}.xlsx" df.to_excel(filename, index=False) speak("تم حفظ التقرير") self.status_text.configure(text="تم الحفظ!", text_color="green") def run_ai_loop(self): ret, frame = self.cap.read() if ret: # معالجة الصورة frame = cv2.flip(frame, 1) image = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB) results = pose.process(image) if results.pose_landmarks: lm = results.pose_landmarks.landmark # جلب الإحداثيات s = [lm[mp_pose.PoseLandmark.RIGHT_SHOULDER.value].x, lm[mp_pose.PoseLandmark.RIGHT_SHOULDER.value].y] e = [lm[mp_pose.PoseLandmark.RIGHT_ELBOW.value].x, lm[mp_pose.PoseLandmark.RIGHT_ELBOW.value].y] w = [lm[mp_pose.PoseLandmark.RIGHT_WRIST.value].x, lm[mp_pose.PoseLandmark.RIGHT_WRIST.value].y] h = [lm[mp_pose.PoseLandmark.RIGHT_HIP.value].x, lm[mp_pose.PoseLandmark.RIGHT_HIP.value].y] # حساب الزاوية حسب التمرين المختار if self.current_exercise == "الكوع": angle = self.calc_angle(s, e, w) else: # الكتف angle = self.calc_angle(h, s, e) # منطق العدّ if angle > 160: if self.stage == "up": self.counter += 1 speak(str(self.counter)) self.history_data.append([datetime.now().strftime("%H:%M:%S"), self.current_exercise, self.counter]) self.stage = "down" if angle < 30: self.stage = "up" # تحديث الشاشة self.angle_text.configure(text=f"الزاوية: {int(angle)}°") self.counter_text.configure(text=str(self.counter)) # رسم النقاط mp_drawing.draw_landmarks(image, results.pose_landmarks, mp_pose.POSE_CONNECTIONS) # عرض الصورة في الواجهة img = Image.fromarray(image) imgtk = ImageTk.PhotoImage(image=img) self.video_display.imgtk = imgtk self.video_display.configure(image=imgtk) self.after(10, self.run_ai_loop) if __name__ == "__main__": app = SmartPhysioFullApp() app.mainloop()
